Numbers Example 7: Math.max()

Check the code below and click the buttons to see the outcomes.


          let maxNumber = Math.max(1, 20, 40, 100);

          alert(maxNumber);
        

          let bunchOfNumbers = [1, 20, 40, 100];

          let maxNumber = Math.max(bunchOfNumbers);

          alert(maxNumber);
        

          let bunchOfNumbers = [1, 20, 40, 100];

          let maxNumber = Math.max(...bunchOfNumbers); //Using the spread operator!

          alert(maxNumber);
        

          let bunchOfNumbers = [1, 20, 40, 100];

          let maxNumber = bunchOfNumbers.reduce(function(a, b){
            return Math.max(a, b);
          })

          alert(maxNumber);
