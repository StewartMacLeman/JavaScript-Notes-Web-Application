Check the code below and click the buttons to see the outcomes.

          alert(date);
        

          let date = new Date();
          let milliSecDay = 24 * 60 * 60 * 1000;
          date.setTime(-milliSecDay);
          alert(date);
        

          let date = new Date();
          let milliSecDay = 24 * 60 * 60 * 1000;
          date.setTime(milliSecDay * 3);
          alert(date);

          date.setTime(date.getTime() + milliSecDay);
          alert(date);
