Function Example 6

Check the code below and click the button to see the outcomes.


          function compareValue(a , b){
            if (a > b){
              alert("a is greater than b!");
            }
            else if (a < b){
              alert("b is greater than a");
            }
            else if (a == b){
              alert("the values are the same");
            }
          };

          compareValue(10, 5); // The first alert!
          compareValue(5, 10); // The second alert!
          compareValue(5, 5);  // The third alert!
